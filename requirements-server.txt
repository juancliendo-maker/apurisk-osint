# Extracción de texto de PDFs adjuntos al análisis de caso.
# Usamos pypdf (sucesor moderno de PyPDF2) en lugar de pdfplumber porque:
# - pypdf: ~3MB, ~50MB RAM en uso (suficiente para texto plano)
# - pdfplumber: ~30MB, ~150MB RAM (arrastra Pillow + pdfminer.six,
#   provoca OOM en plan Render Starter de 512MB)
pypdf==5.1.0

# LLM client para Executive Synthesis Engine (narrativa estratégica).
# Modelo default: claude-haiku-4-5 (rápido y barato para narrativas cortas).
# Se activa SOLO si ANTHROPIC_API_KEY está en env vars; si no, fallback
# determinístico por plantillas (cero costo, narrativa más robotica pero
# funcional). Esto preserva el deploy aunque la key esté ausente o expire.
#
# IMPORTANTE: requerir anthropic>=0.49.0 porque versiones anteriores
# pasan 'proxies=' al constructor de httpx.Client(), parámetro que
# httpx>=0.28 ya no acepta (rompe con TypeError en runtime).
anthropic>=0.49.0,<1.0.0
